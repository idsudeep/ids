
							Ajax Format
							
						$.ajax({
						type: "POST",
						url: Url,
						data: dataParameter,
						async: true,
						success: function(results, textStatus) {
							debugger;
							console.log("success : " + results);
						},
						error: function(xhr, status, error)
						{
							console.log("error : " + xhr.responseText);                   
						}
						}).done(function(results) {
							debugger;          
							console.log("done : " + results);           
						});
